    while true {
        // Block until "calibration complete" or "get_distance" is received
        receive_radio_data("get_distance(x, y)");

        if (received_command == "calibration_complete") {
            send_radio_data("ACK");
            // Calibration is complete, exit program
            return;
        }

        // Here, we assume that the received command is not "calibration complete"
        // Parse the command to determine if its get_distance. If so, is this devices ID the first or second argument?
        parse_command();

            continue;
